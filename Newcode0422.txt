// ================= ADC SETTINGS =================
static const float ADC_REF = 3.3;
static const int ADC_MAX = 4095;

// Adjust after measuring the actual battery divider
static const float BATTERY_DIVIDER_RATIO = 2.0;

// Adjust after measuring dry and wet soil values on your actual board
static const int SOIL_ADC_DRY = 3200;
static const int SOIL_ADC_WET = 1200;

// ================= PH SETTINGS =================
float voltage_pH7 = 0.477;
float voltage_pH4 = 0.512;

const int phSampleCount = 30;
const float phFilterAlpha = 0.1;
float filteredPH = 7.0;

// ================= HELPER FUNCTIONS =================
float adcToVoltage(int raw) {
  return (raw * ADC_REF) / ADC_MAX;
}

float constrainFloat(float value, float minVal, float maxVal) {
  if (value < minVal) return minVal;
  if (value > maxVal) return maxVal;
  return value;
}

float mapFloat(float x, float in_min, float in_max, float out_min, float out_max) {
  if (in_max == in_min) return out_min;
  return (x - in_min) * (out_max - out_min) / (in_max - in_min) + out_min;
}

float batteryVoltageToPercent(float voltage) {
  float percent = mapFloat(voltage, 3.20, 4.20, 0.0, 100.0);
  return constrainFloat(percent, 0.0, 100.0);
}

float soilAdcToPercent(int raw) {
  float percent = mapFloat((float)raw, (float)SOIL_ADC_DRY, (float)SOIL_ADC_WET, 0.0, 100.0);
  return constrainFloat(percent, 0.0, 100.0);
}

float readPHVoltage() {
  uint32_t total = 0;

  for (int i = 0; i < phSampleCount; i++) {
    total += analogRead(PH_PIN);
    delay(10);
  }

  float raw = total / (float)phSampleCount;
  rawPhADC = raw;
  float voltage = (raw / ADC_MAX) * ADC_REF;
  return voltage;
}

float phVoltageToValue(float voltage) {
  float slope = (7.0 - 4.0) / (voltage_pH7 - voltage_pH4);
  float intercept = 7.0 - (slope * voltage_pH7);
  float phValue = (slope * voltage) + intercept;
  return phValue * cal_ph;
}

float tdsVoltageToValue(float voltage) {
  return (voltage * 500.0) * cal_tds;
}

// ================= HTTP HANDLERS =================
void handleRoot() {
  if (isAuthenticated) {
    server.sendHeader("Location", "/dashboard", true);
    server.send(302, "text/plain", "");
  } else {
    server.send(200, "text/html", loginPage(""));
  }
}

void handleLogin() {
  if (server.method() == HTTP_POST) {
    String u = server.arg("username");
    String p = server.arg("password");

    if (u == login_username && p == login_password) {
      isAuthenticated = true;
      server.sendHeader("Location", "/dashboard", true);
      server.send(302, "text/plain", "");
    } else {
      server.send(200, "text/html", loginPage("Invalid credentials!"));
    }
  } else {
    server.send(405, "text/plain", "Method Not Allowed");
  }
}

void handleDashboard() {
  if (!isAuthenticated) {
    server.sendHeader("Location", "/", true);
    server.send(302, "text/plain", "");
    return;
  }
  server.send(200, "text/html", dashboardPage());
}

void handleCalibration() {
  if (!isAuthenticated) {
    server.sendHeader("Location", "/", true);
    server.send(302, "text/plain", "");
    return;
  }

  if (server.method() == HTTP_POST) {
    if (server.hasArg("cal_batteryLevel")) cal_batteryLevel = server.arg("cal_batteryLevel").toFloat();
    if (server.hasArg("cal_ph")) cal_ph = server.arg("cal_ph").toFloat();
    if (server.hasArg("cal_tds")) cal_tds = server.arg("cal_tds").toFloat();
    if (server.hasArg("cal_liquidTemp")) cal_liquidTemp = server.arg("cal_liquidTemp").toFloat();
    if (server.hasArg("cal_airTemp")) cal_airTemp = server.arg("cal_airTemp").toFloat();
    if (server.hasArg("cal_humidity")) cal_humidity = server.arg("cal_humidity").toFloat();
    if (server.hasArg("cal_universalPh")) cal_universalPh = server.arg("cal_universalPh").toFloat();
    if (server.hasArg("cal_universalTds")) cal_universalTds = server.arg("cal_universalTds").toFloat();
    if (server.hasArg("cal_soilMoisture")) cal_soilMoisture = server.arg("cal_soilMoisture").toFloat();
    if (server.hasArg("cal_co2")) cal_co2 = server.arg("cal_co2").toFloat();
    if (server.hasArg("cal_baroPressure")) cal_baroPressure = server.arg("cal_baroPressure").toFloat();
    if (server.hasArg("cal_rgb_r")) cal_rgb_r = server.arg("cal_rgb_r").toFloat();
    if (server.hasArg("cal_rgb_g")) cal_rgb_g = server.arg("cal_rgb_g").toFloat();

void handleLogout() {
  isAuthenticated = false;

// ================= SENSOR UPDATE =================
void updateSensors() {
  float h = dht.readHumidity();
  float t = dht.readTemperature();

  if (!isnan(h) && !isnan(t)) {
    humidity = h * cal_humidity;
    airTemp = t * cal_airTemp;
  }

  float phVoltage = readPHVoltage();
  float phValue = phVoltageToValue(phVoltage);
  filteredPH = (1.0 - phFilterAlpha) * filteredPH + (phFilterAlpha * phValue);
  ph = filteredPH;

  int rawTds = analogRead(TDS_PIN);
  int rawSoil = analogRead(MOISTURE_SENSOR_PIN);
  int rawBattery = analogRead(BSENS_PIN);

  rawTdsADC = rawTds;
  rawSoilADC = rawSoil;
  rawBatteryADC = rawBattery;

  float tdsVoltage = adcToVoltage(rawTds);
  batteryAdcVoltage = adcToVoltage(rawBattery);

  tds = tdsVoltageToValue(tdsVoltage);

  soilMoisture = soilAdcToPercent(rawSoil) * cal_soilMoisture;
  soilMoisture = constrainFloat(soilMoisture, 0.0, 100.0);

  batteryVoltage = batteryAdcVoltage * BATTERY_DIVIDER_RATIO;
  batteryLevel = batteryVoltageToPercent(batteryVoltage) * cal_batteryLevel;
  batteryLevel = constrainFloat(batteryLevel, 0.0, 100.0);

  universalPh = ph * cal_universalPh;
  universalTds = tds * cal_universalTds;

  waterTempSensor.requestTemperatures();
  float tempC = waterTempSensor.getTempCByIndex(0);

  if (tempC != DEVICE_DISCONNECTED_C) {
    liquidTemp = tempC;
  } else {
    liquidTemp = 0.0;
  }
}

  Serial.print("Liquid Temp: "); Serial.print(liquidTemp); Serial.println(" C");
  Serial.print("Soil Moisture: "); Serial.print(soilMoisture); Serial.println(" %");
  Serial.print("Battery Voltage: "); Serial.print(batteryVoltage, 3); Serial.println(" V");
  Serial.print("Battery Level: "); Serial.print(batteryLevel); Serial.println(" %");
  Serial.print("Raw pH ADC: "); Serial.println(rawPhADC);
  Serial.print("Raw TDS ADC: "); Serial.println(rawTdsADC);
  Serial.print("Raw Soil ADC: "); Serial.println(rawSoilADC);
  Serial.print("Raw Battery ADC: "); Serial.println(rawBatteryADC);
  Serial.println("==========================================");
}

// ================= SETUP =================
void setup() {
  Serial.begin(115200);

  pinMode(ENABLE_PIN, OUTPUT);
  digitalWrite(ENABLE_PIN, HIGH);
  Serial.println("Moisture Sensor Powered On.");

  pinMode(PH_PIN, INPUT);
  pinMode(TDS_PIN, INPUT);
  pinMode(MOISTURE_SENSOR_PIN, INPUT);
  pinMode(BSENS_PIN, INPUT);

  analogReadResolution(12);
  analogSetPinAttenuation(PH_PIN, ADC_11db);

  delay(500);
  dht.begin();
  waterTempSensor.begin();

  WiFi.softAP(ssid, password);
  MDNS.begin(mdnsName);

  server.on("/", HTTP_GET, handleRoot);
  server.on("/login", HTTP_POST, handleLogin);
  server.on("/dashboard", HTTP_GET, handleDashboard);
  server.on("/calibration", HTTP_GET, handleCalibration);
  server.on("/calibration", HTTP_POST, handleCalibration);
  server.on("/logout", HTTP_GET, handleLogout);
  server.begin();

  Serial.println("System Active");
  Serial.println("GPIO19 sensor rail enabled");
  Serial.println("GPIO27 DHT11 initialized");
  Serial.println("GPIO18 water temp initialized");
  Serial.println("GPIO36 pH sensor initialized with 2 point calibration");
}

// ================= LOOP =================
void loop() {
  server.handleClient();

  static unsigned long lastMeasure = 0;
  if (millis() - lastMeasure >= 2000) {
    lastMeasure = millis();
    updateSensors();
    printSensors();
  }
}
